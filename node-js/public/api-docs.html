PHP Example:
<?php $url = 'http://localhost:8080/compress'; $file_path = 'video.mp4'; $post_data = [ 'file' => new CURLFile($file_path), 'speed' => 'ai' ]; $ch = curl_init(); curl_setopt($ch, CURLOPT_URL, $url); curl_setopt($ch, CURLOPT_POST, true); curl_setopt($ch, CURLOPT_POSTFIELDS, $post_data); curl_setopt($ch, CURLOPT_RETURNTRANSFER, true); curl_setopt($ch, CURLOPT_HTTPHEADER, [ 'Accept: application/octet-stream' ]); $response = curl_exec($ch); $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE); curl_close($ch); if ($http_code === 200) { file_put_contents('compressed.mp4', $response); echo "Video compressed successfully!"; } else { echo "Error: HTTP $http_code"; } ?>
Laravel Example:
<?php namespace App\Http\Controllers; use Illuminate\Http\Request; use Illuminate\Support\Facades\Http; use Illuminate\Support\Facades\Storage; class VideoController extends Controller { public function compress(Request $request) { $request->validate([ 'video' => 'required|file|mimes:mp4,mov,avi,mkv|max:2048' ]); $file = $request->file('video'); $speed = $request->input('speed', 'ai'); $response = Http::attach( 'file', file_get_contents($file->getPathname()), $file->getClientOriginalName() )->post('http://localhost:8080/compress', [ 'speed' => $speed ])->withHeaders([ 'Accept' => 'application/octet-stream' ]); if ($response->successful()) { $compressed_path = 'compressed/' . time() . '_compressed.mp4'; Storage::put($compressed_path, $response->body()); return response()->json([ 'success' => true, 'file' => $compressed_path, 'headers' => $response->headers() ]); } return response()->json([ 'success' => false, 'error' => 'Compression failed' ], 500); } } ?>
Node.js Example:
const FormData = require('form-data'); const fs = require('fs'); const fetch = require('node-fetch'); const form = new FormData(); form.append('file', fs.createReadStream('video.mp4')); form.append('speed', 'ai'); const response = await fetch('http://localhost:8080/compress', { method: 'POST', body: form, headers: { 'Accept': 'application/octet-stream' } }); const fileStream = fs.createWriteStream('compressed.mp4'); response.body.pipe(fileStream);

🔌 API Endpoints

Complete API reference with examples and parameters

GET /health
Check server status and get available features
Response:
{ "ok": true, "service": "videocompress", "version": "3.2.0-orientation", "modes": ["ai", "turbo", "max", "ultra_fast", "super_fast", "fast", "balanced", "quality"], "defaults": { "codec": "h264", "resolution": "original", "hw": "none" } }
POST /compress
Compress a video file with specified settings

Parameters:

Parameter Type Required Default Description
file File Required - Video file to compress
speed String Optional ai Compression speed mode
resolution String Optional original Output resolution
codec String Optional h264 Video codec
audio String Optional aac Audio codec
hw String Optional none Hardware acceleration

Response Headers (API Mode):

Header Description Example
X-Mode Final compression mode used balanced
X-Encode-Duration-Ms Encoding time in milliseconds 15000
X-Throughput-MBps Processing speed in MB/s 25.5
X-Input-Bytes Original file size 52428800
X-Output-Bytes Compressed file size 15728640
Example Request:
curl -X POST \ -H "Accept: application/octet-stream" \ -F "file=@video.mp4" \ -F "speed=balanced" \ -F "resolution=720p" \ -F "codec=h264" \ -F "hw=videotoolbox" \ -o compressed.mp4 \ http://localhost:8080/compress
GET /dl/{id}
Download a compressed file from the temporary store (web UI flow only)

Parameters:

Parameter Type Required Description
id String Required File ID from compression result
name String Optional Custom filename for download
GET /meta/{id}
Get metadata for a compressed file (web UI flow only)
Response:
{ "id": "abc123def456", "mode": "balanced", "mode_decider": "ai", "input_bytes": 52428800, "output_bytes": 15728640, "resolution": "720p", "codec": "h264", "audio": "aac", "hw": "none", "encode_duration_ms": 15000, "throughput_mb_s": 25.5 }

Basic PHP with cURL:
<?php function compressVideo($file_path, $speed = 'ai', $api_url = 'http://localhost:8080/compress') { if (!file_exists($file_path)) { throw new Exception("File not found: $file_path"); } $post_data = [ 'file' => new CURLFile($file_path), 'speed' => $speed, 'resolution' => '720p' ]; $ch = curl_init(); curl_setopt_array($ch, [ CURLOPT_URL => $api_url, CURLOPT_POST => true, CURLOPT_POSTFIELDS => $post_data, CURLOPT_RETURNTRANSFER => true, CURLOPT_HTTPHEADER => ['Accept: application/octet-stream'], CURLOPT_TIMEOUT => 300 // 5 minutes ]); $response = curl_exec($ch); $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE); $error = curl_error($ch); curl_close($ch); if ($error) { throw new Exception("cURL Error: $error"); } if ($http_code !== 200) { throw new Exception("HTTP Error: $http_code"); } return $response; } // Usage try { $compressed_video = compressVideo('input.mp4', 'balanced'); file_put_contents('output.mp4', $compressed_video); echo "Video compressed successfully!"; } catch (Exception $e) { echo "Error: " . $e->getMessage(); } ?>
PHP OOP Class:
<?php class VideoCompressor { private $api_url; private $timeout; public function __construct($api_url = 'http://localhost:8080/compress', $timeout = 300) { $this->api_url = $api_url; $this->timeout = $timeout; } public function compress($file_path, $options = []) { $defaults = [ 'speed' => 'ai', 'resolution' => 'original', 'codec' => 'h264', 'hw' => 'none' ]; $options = array_merge($defaults, $options); if (!file_exists($file_path)) { throw new InvalidArgumentException("File not found: $file_path"); } $post_data = [ 'file' => new CURLFile($file_path) ]; // Add options to post data foreach ($options as $key => $value) { $post_data[$key] = $value; } $ch = curl_init(); curl_setopt_array($ch, [ CURLOPT_URL => $this->api_url, CURLOPT_POST => true, CURLOPT_POSTFIELDS => $post_data, CURLOPT_RETURNTRANSFER => true, CURLOPT_HTTPHEADER => ['Accept: application/octet-stream'], CURLOPT_TIMEOUT => $this->timeout ]); $response = curl_exec($ch); $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE); $error = curl_error($ch); curl_close($ch); if ($error) { throw new RuntimeException("cURL Error: $error"); } if ($http_code !== 200) { throw new RuntimeException("HTTP Error: $http_code"); } return $response; } public function compressToFile($input_path, $output_path, $options = []) { $compressed_data = $this->compress($input_path, $options); return file_put_contents($output_path, $compressed_data); } } // Usage $compressor = new VideoCompressor(); try { $compressor->compressToFile('input.mp4', 'output.mp4', [ 'speed' => 'balanced', 'resolution' => '720p' ]); echo "Video compressed successfully!"; } catch (Exception $e) { echo "Error: " . $e->getMessage(); } ?>
Laravel Controller:
<?php namespace App\Http\Controllers; use Illuminate\Http\Request; use Illuminate\Support\Facades\Http; use Illuminate\Support\Facades\Storage; use Illuminate\Support\Facades\Log; class VideoController extends Controller { private $api_url = 'http://localhost:8080/compress'; public function compress(Request $request) { $request->validate([ 'video' => 'required|file|mimes:mp4,mov,avi,mkv,webm|max:2048', 'speed' => 'string|in:ai,turbo,max,ultra_fast,super_fast,fast,balanced,quality', 'resolution' => 'string|in:original,360p,480p,720p,1080p,1440p,2160p', 'codec' => 'string|in:h264,h265,copy', 'hw' => 'string|in:none,videotoolbox' ]); try { $file = $request->file('video'); $options = $request->only(['speed', 'resolution', 'codec', 'hw']); // Generate unique filename $filename = time() . '_' . uniqid() . '_compressed.mp4'; $output_path = 'compressed/' . $filename; $response = Http::timeout(300)->attach( 'file', file_get_contents($file->getPathname()), $file->getClientOriginalName() )->post($this->api_url, $options)->withHeaders([ 'Accept' => 'application/octet-stream' ]); if ($response->successful()) { Storage::put($output_path, $response->body()); return response()->json([ 'success' => true, 'file' => $output_path, 'original_size' => $file->getSize(), 'compressed_size' => Storage::size($output_path), 'compression_ratio' => round((1 - Storage::size($output_path) / $file->getSize()) * 100, 2), 'headers' => $response->headers() ]); } Log::error('Video compression failed', [ 'status' => $response->status(), 'body' => $response->body() ]); return response()->json([ 'success' => false, 'error' => 'Compression failed' ], 500); } catch (\Exception $e) { Log::error('Video compression error', [ 'error' => $e->getMessage(), 'file' => $file->getClientOriginalName() ]); return response()->json([ 'success' => false, 'error' => 'Internal server error' ], 500); } } public function download($filename) { $path = 'compressed/' . $filename; if (!Storage::exists($path)) { return response()->json(['error' => 'File not found'], 404); } return Storage::download($path); } } ?>
Laravel Service Class:
<?php namespace App\Services; use Illuminate\Support\Facades\Http; use Illuminate\Support\Facades\Storage; use Illuminate\Support\Facades\Log; class VideoCompressionService { private $api_url; private $timeout; public function __construct($api_url = null, $timeout = 300) { $this->api_url = $api_url ?? config('services.videocompress.url', 'http://localhost:8080/compress'); $this->timeout = $timeout; } public function compress($file_path, $options = []) { $defaults = [ 'speed' => 'ai', 'resolution' => 'original', 'codec' => 'h264', 'hw' => 'none' ]; $options = array_merge($defaults, $options); if (!file_exists($file_path)) { throw new \InvalidArgumentException("File not found: $file_path"); } try { $response = Http::timeout($this->timeout)->attach( 'file', file_get_contents($file_path), basename($file_path) )->post($this->api_url, $options)->withHeaders([ 'Accept' => 'application/octet-stream' ]); if (!$response->successful()) { throw new \RuntimeException("Compression failed: HTTP " . $response->status()); } return [ 'data' => $response->body(), 'headers' => $response->headers(), 'size' => strlen($response->body()) ]; } catch (\Exception $e) { Log::error('Video compression service error', [ 'file' => $file_path, 'options' => $options, 'error' => $e->getMessage() ]); throw $e; } } public function compressAndSave($input_path, $output_path, $options = []) { $result = $this->compress($input_path, $options); $directory = dirname($output_path); if (!is_dir($directory)) { mkdir($directory, 0755, true); } file_put_contents($output_path, $result['data']); return [ 'output_path' => $output_path, 'size' => $result['size'], 'headers' => $result['headers'] ]; } public function compressToStorage($input_path, $storage_path, $options = []) { $result = $this->compress($input_path, $options); Storage::put($storage_path, $result['data']); return [ 'storage_path' => $storage_path, 'size' => $result['size'], 'headers' => $result['headers'] ]; } } // Usage in Controller class VideoController extends Controller { private $compressionService; public function __construct(VideoCompressionService $compressionService) { $this->compressionService = $compressionService; } public function compress(Request $request) { $file = $request->file('video'); $options = $request->only(['speed', 'resolution', 'codec', 'hw']); $output_path = 'compressed/' . time() . '_compressed.mp4'; try { $result = $this->compressionService->compressToStorage( $file->getPathname(), $output_path, $options ); return response()->json([ 'success' => true, 'file' => $output_path, 'size' => $result['size'] ]); } catch (\Exception $e) { return response()->json([ 'success' => false, 'error' => $e->getMessage() ], 500); } } } ?>
